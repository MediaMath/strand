strand-checkbox

A fully-styled checkbox.

Example HTML Editor

<!-- Defaults Example --> <strand-checkbox></strand-checkbox> <strand-checkbox> <label>Checkbox With Label</label> </strand-checkbox> <strand-checkbox state="partial"></strand-checkbox> <strand-checkbox checked="true" validation="checked"></strand-checkbox> <strand-checkbox checked="true" disabled></strand-checkbox> <!-- Icon Example --> <style> .check-cbx { color: #008000; } </style> <strand-checkbox icon="chart"></strand-checkbox> <strand-checkbox class="check-cbx" checked icon="check"> <label>Check me out</label> </strand-checkbox>
