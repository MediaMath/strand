mm-popover

A fully-styled popover

Example HTML Editor

<mm-button id="cpb" unresolved> <label>open popover</label> </mm-button> <mm-popover id="popover" target="#cpb" unresolved> <div class="body"> <mm-input></mm-input> </div> <mm-footer> <mm-action id="closeLink" underline> <label>Close</label> </mm-action> <mm-button id="applyBtn"> <label>Apply</label> </mm-button> </mm-footer> </mm-popover> <!-- script to trigger the popover --> <script> var cpb, popover, closeLink; window.addEventListener("polymer-ready", function() { cpb = document.getElementById("cpb"); popover = document.getElementById("popover"); closeLink = document.getElementById("closeLink"); cpb.addEventListener("click", cpbClickHandler); closeLink.addEventListener("click", closeLinkClickHandler); }); function cpbClickHandler(e) { popover.toggle(); } function closeLinkClickHandler(e){ e.preventDefault(); popover.toggle(); } </script>
