mm-footer

A fully-styled form footer - for use within a form footer or popover component. Capable of displaying variable button set and feedback messages.

Example HTML Editor

<!-- Default Example --> <mm-footer unresolved fitparent> <mm-button> <label>Save</label> </mm-button> </mm-footer> <!-- Display Message / Multiple Buttons Example --> <mm-footer id="tf1" unresolved fitparent> <mm-action id="closeLink" underline="true"> <label>Close</label> </mm-action> <mm-button id="saveAndClose"> <label>Save &amp; Close</label> </mm-button> <mm-button id="saveAndAdd"> <label>Save &amp; Add Another</label> </mm-button> </mm-footer> <!-- Align Items Left or Right Example --> <mm-footer id="tf2" unresolved fitparent> <mm-checkbox class="left"> <label>Don't show again</label> </mm-checkbox> <mm-button> <label>Close</label> </mm-button> </mm-footer> <!-- Example Scripts --> <script> var tf1, tf2, closeLink, saveAndClose, saveAndAdd; window.addEventListener("WebComponentsReady", function() { tf1 = document.querySelector("#tf1"); tf2 = document.querySelector("#tf2"); closeLink = document.querySelector("#closeLink"); saveAndClose = document.querySelector("#saveAndClose"); saveAndAdd = document.querySelector("#saveAndAdd"); closeLink.addEventListener("click", closeHandler); saveAndClose.addEventListener("click", saveCloseHandler); saveAndAdd.addEventListener("click", saveAddHandler); }); function closeHandler(e) { e.preventDefault(); console.log("closeHandler"); // Test messaging: tf1.message = "This is a neat message."; tf1.type = "success"; tf1.showMessage(); } function saveCloseHandler(e) { console.log("saveCloseHandler"); // Test messaging: tf1.message = "This is a super message."; tf1.type = "error"; tf1.showMessage(); } function saveAddHandler(e) { console.log("saveAddHandler"); // Test messaging: tf1.message = "This is a test message."; tf1.type = "warning"; tf1.showMessage(); } </script>
