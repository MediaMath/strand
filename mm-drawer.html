mm-drawer

Re-usable drawer component for showing/hiding content.

Example HTML Editor

<style> mm-drawer { width: 100%; } #drawer1 { padding-top: 5px; } #drawer2 { margin: 30px 0 10px; } .container { padding: 10px; } </style> <!-- Tab style Drawer example --> <template is="dom-bind"> <mm-drawer id="drawer1" expanded="{{expanded}}" opened-label="Hide Drawer" closed-label="Show Drawer"> <div class="container"> <h2>Nothing to see here...</h2> </div> </mm-drawer> </template> <!-- Inset style Drawer example --> <template is="dom-bind"> <mm-drawer id="drawer2" expanded="{{checked}}" closedHeight="45"> <div class="container"> <mm-checkbox checked="{{checked}}" id="checkbox"><label>Toggle Inset Drawer</label></mm-checkbox> <h2>Well Hello.</h2> </div> </mm-drawer> <div> <mm-button id="toggleBtn"> <label>Toggle Inset Drawer</label> </mm-button> <mm-button id="hideBtn"> <label>Hide Inset Drawer</label> </mm-button> </div> <script type="text/javascript"> var drawer2 = document.querySelector("#drawer2"); var toggleBtn = document.querySelector("#toggleBtn"); var hideBtn = document.querySelector("#hideBtn"); toggleBtn.onclick = function(){ drawer2.closedHeight = 45; drawer2.toggle(); } hideBtn.onclick = function(){ drawer2.closedHeight = 0; drawer2.close(); } </script> </template>

default : none
options : none
attribute : opened-label
reflected to attribute : no 		Sets the label text which appears on the drawer's tab button when drawer is opened. Setting this value will show the 'tab' style drawer. If not set, and 'inset' style drawer will be displayed.
closedLabel : String (optional) 
default : none
options : none
attribute : closed-label
reflected to attribute : no 		Sets the label text which appears on the drawer's tab button when the drawer is closed. Setting this value will show the 'tab' style drawer. If not set, and 'inset' style drawer will be displayed.
openedHeight : Number (optional) 
default : *height of contents
options : none
attribute : opened-height
reflected to attribute : no 		Sets the pixel height of the drawer, when opened. If not set, the opened height of the drawer will he the measured height of the content contained within.
closedHeight : Number (optional) 
default : 0
options : none
attribute : closed-height
reflected to attribute : no 		Sets the pixel height of the drawer, when closed. If not set, the closed drawer will be zero pixels tall.
forceMeasure : Boolean (optional) 
default : false
options : none
attribute : force-measure
reflected to attribute : no 		If set, forces the drawer to measure it's contents every time it is opened. By default, the drawer's initial height is cached (see: openedHeight).
expanded : Boolean (optional) 
default : false
options : none
attribute : expanded
reflected to attribute : no 		Sets or returns the expanded state of the drawer. Changing the value will open or close the drawer.
