strand-grid-item

A fully styled, configurable data grid item.

Example HTML Editor

<strand-grid id="mmTestGrid" unresolved style="width: 100%; height: 400px;"> <strand-grid-column field="first_name" resize sort>First Name</strand-grid-column> <strand-grid-column field="last_name" resize sort>Last Name</strand-grid-column> <strand-grid-column field="actions" sort></strand-grid-column> <template preserve-content> <strand-grid-item model="{{model}}" scope="{{scope}}"> <div field="first_name">Custom Text - <span>{{model.first_name}}</span></div> <div field="actions"><strand-icon width="15" height="15" type="actions"></strand-icon></div> </strand-grid-item> </template> </strand-grid> <script> var grid = document.querySelector("#mmTestGrid"), data = []; for(var i=0; i<50; i++){ data.push({ first_name: "First", last_name: "Last", email: "text@example.com" }); } window.addEventListener("WebComponentsReady", function() { grid.data = data; }); </script>
