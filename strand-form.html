strand-form

A form component for creating visual layout, collecting and validating data from the individual fields placed within.

This component contains experimental features. The configuration and API are subject to change. Please use at your own risk.

Example HTML Editor

<strand-form id="testForm" columns="4" footer-fixed="false"> <div class="col" span="1"> <strand-input fitparent name="input" placeholder="Type a Number"></strand-input> </div> <div class="col" span="1"> <strand-dropdown fitparent name="dropdown" placeholder="Select an Item"> <strand-list-item>List Item 1</strand-list-item> <strand-list-item>List Item 2</strand-list-item> <strand-list-item>List Item 3</strand-list-item> <strand-list-item>List Item 4</strand-list-item> </strand-dropdown> </div> <div class="col" span="2"> <strand-group fitparent name="radio"> <strand-radio> <label>Red</label> </strand-radio> <strand-radio> <label>Green</label> </strand-radio> <strand-radio> <label>Blue</label> </strand-radio> </strand-group> </div> </strand-form> <script> var config = { 'input' : { validation: 'int|empty', parentEle: null, errorMsgEle: null, errorMsg: 'You need to type a number', label: 'Type a number' }, 'dropdown' : { validation: 'empty', parentEle: null, errorMsg: 'You need to select an item', label: 'Select an Item' }, 'radio' : { validation: function(name, value, data) { return data[name] === 'Red' && value === 'Red'; }, parentEle: null, errorMsgEle: null, errorMsg: 'You need to select \'Red\'', label: 'Select a Color' } }; window.addEventListener("WebComponentsReady", function() { var testForm = document.querySelector('#testForm'); testForm.config = config; }); </script>
